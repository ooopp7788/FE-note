function reactive(target: object) {
  if (isRef(target)) {
    return target
  }
  return createReactiveObject(
    target,
    rawToReactive,
    reactiveToRaw,
    mutableHandlers, // 普通属性 Proxy options -> baseHandler.ts
    mutableCollectionHandlers // 特殊类型(Set, Map, WeakMap, WeakSet)时 Proxy options -> collectionHandler.ts
  )
}

function createReactiveObject(
  target: unknown,
  toProxy: WeakMap<any, any>,
  toRaw: WeakMap<any, any>,
  baseHandlers: ProxyHandler<any>,
  collectionHandlers: ProxyHandler<any>
) {
  if (!isObject(target)) {
    if (__DEV__) {
      console.warn(`value cannot be made reactive: ${String(target)}`)
    }
    return target
  }
  // target already has corresponding Proxy
  let observed = toProxy.get(target)
  if (observed !== void 0) {
    return observed
  }
  // target is already a Proxy
  if (toRaw.has(target)) {
    return target
  }
  // only a whitelist of value types can be observed.
  if (!canObserve(target)) {
    return target
  }
  const handlers = collectionTypes.has(target.constructor)
    ? collectionHandlers
    : baseHandlers
  observed = new Proxy(target, handlers)
  // 双向记录 target Proxy 关系
  toProxy.set(target, observed)
  toRaw.set(observed, target)
  return observed
}

// 以一般情况为例
function createGetter(isReadonly = false, shallow = false) {
  return function get(target: object, key: string | symbol, receiver: object) {
    if (shallow) {
      track(target, TrackOpTypes.GET, key)
      return res
    }
    // ref unwrapping, only for Objects, not for Arrays.
    if (isRef(res) && !isArray(target)) {
      return res.value
    }
    // 依赖收集
    track(target, TrackOpTypes.GET, key)
    return isObject(res)
      ? isReadonly
        ? // need to lazy access readonly and reactive here to avoid
          // circular dependency
          readonly(res)
        // 在触发 getter 时, 再递归 reactive nested object
        // 并不会在初始化时就 递归定义 reactive
        : reactive(res)
      : res
  }
}

function createSetter(isReadonly = false, shallow = false) {
  return function set(
    target: object,
    key: string | symbol,
    value: unknown,
    receiver: object
  ): boolean {
    if (isReadonly && LOCKED) {
      return true
    }

    const oldValue = (target as any)[key]
    if (!shallow) {
      value = toRaw(value)
      if (!isArray(target) && isRef(oldValue) && !isRef(value)) {
        oldValue.value = value
        return true
      }
    } else {
      // in shallow mode, objects are set as-is regardless of reactive or not
    }

    const hadKey = hasOwn(target, key)
    const result = Reflect.set(target, key, value, receiver)
    // don't trigger if target is something up in the prototype chain of original
    if (target === toRaw(receiver)) {
      if (!hadKey) {
        trigger(target, TriggerOpTypes.ADD, key, value)
      } else if (hasChanged(value, oldValue)) {
        trigger(target, TriggerOpTypes.SET, key, value, oldValue)
      }
    }
    return result
  }
}

function computed<T>(
  getterOrOptions: ComputedGetter<T> | WritableComputedOptions<T>
) {
  let getter: ComputedGetter<T>
  let setter: ComputedSetter<T>

  let dirty = true
  let value: T
  let computed: ComputedRef<T>

  const runner = effect(getter, {
    lazy: true,
    // mark effect as computed so that it gets priority during trigger
    computed: true,
    scheduler: () => {
      if (!dirty) {
        dirty = true
        trigger(computed, TriggerOpTypes.SET, 'value')
      }
    }
  })
  computed = {
    _isRef: true,
    // expose effect so computed can be stopped
    effect: runner,
    get value() {
      if (dirty) {
        value = runner()
        dirty = false
      }
      track(computed, TrackOpTypes.GET, 'value')
      return value
    },
    set value(newValue: T) {
      setter(newValue)
    }
  } as any
  return computed
}

> effect.ts

```TypeScript
function effect<T = any>(
  fn: () => T,
  options: ReactiveEffectOptions = EMPTY_OBJ
): ReactiveEffect<T> {
  if (isEffect(fn)) {
    fn = fn.raw
  }
  // effect 是一个函数, 相当于 2.0 里的 watcher
  const effect = createReactiveEffect(fn, options)
  // 和 2.0 一致, computed 属性 lazy: true
  // 其他默认调用一次 effect, 收集一次依赖
  if (!options.lazy) {
    effect()
  }
  return effect
}

function createReactiveEffect<T = any>(
  fn: () => T,
  options: ReactiveEffectOptions
): ReactiveEffect<T> {
    // 返回 effect 函数
  const effect = function reactiveEffect(...args: unknown[]): unknown {
      // 实际执行 run, 收集依赖
    return run(effect, fn, args)
  } as ReactiveEffect
  effect._isEffect = true
  effect.active = true
  effect.raw = fn
  // 依赖的 reactive 属性
  effect.deps = []
  effect.options = options
  return effect
}

function run(effect: ReactiveEffect, fn: Function, args: unknown[]): unknown {
  if (!effect.active) {
    return fn(...args)
  }
  if (!effectStack.includes(effect)) {
    cleanup(effect) // 清理旧依赖
    try {
      enableTracking()
      effectStack.push(effect)
      activeEffect = effect
      return fn(...args)
    } finally {
      effectStack.pop()
      resetTracking()
      activeEffect = effectStack[effectStack.length - 1]
    }
  }
}

// reactive 属性 getter 中会调用, 用于收集当前 effect 依赖
function track(target: object, type: TrackOpTypes, key: unknown) {
  if (!shouldTrack || activeEffect === undefined) {
    return
  }
  // targetMap 用于记录当前 target 需要触发的 effects
  // 在 target Proxy setter 中会调用 trigger 方法触发 effects
    // targetMap:WeakMap = {
    //     [key:typeof target]: <Map>{
    //         [key: keyof target]: Set[effect]
    //     }
    // }
  let depsMap = targetMap.get(target)
  if (depsMap === void 0) {
    targetMap.set(target, (depsMap = new Map()))
  }
  let dep = depsMap.get(key)
  if (dep === void 0) {
    depsMap.set(key, (dep = new Set()))
  }
  if (!dep.has(activeEffect)) {
    dep.add(activeEffect)
    activeEffect.deps.push(dep)
    if (__DEV__ && activeEffect.options.onTrack) {
      activeEffect.options.onTrack({
        effect: activeEffect,
        target,
        type,
        key
      })
    }
  }
}
```


#### 总结
1. 采用 Proxy 替代 Object.defineProperty, 数组操作不需要 hack 实现
2. 完全放弃了 class Watcher Dep, 全部采用函数实现
3. 新增 effect 概念, 对应旧版 Watcher, 所有 reactive 拦截操作都被称为 effect, 表示此次赋值的副作用, , 但原理和 2.0 一致
4. 和 2.0 不同的是, reactive 并不是初始化时递归处理 nested object, 而是在触发 getter 时做包装 object 值, 不触发 getter 的值永远不会做包装, 是一种 lazy 优化
5. 新增 ref 值包裹方式, 类似 mobx 里的 Observer.box, 但在 setup 内调用时, 需要使用 someRef.value 来做取值, 模板中直接使用 {{ someRef }} 就可以
